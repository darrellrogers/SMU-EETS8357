1. #!/usr/bin/env bash set -euo pipefail --- Prereqs and base updates ---sudo apt-get update sudo apt-get upgrade -y Root shell (optional for interactive runs)sudo su--Core packagessudo apt-get install -y docker.io conntrack curl wget git tar --Install kubectl (snap classic) If snap isn't available in your environment, you can skip or install via apt from Kubernetes repo.if command -v snap >/dev/null 2>&1; then sudo snap install kubectl --classic fi --- Minikube (replace deprecated GCS link) -----Download latest minikube amd64 binary and installcurl -LO https://github.com/kubernetes/minikube/releases/latest/download/minikube-linux-amd64 chmod 755 minikube-linux-amd64 sudo mv minikube-linux-amd64 /usr/local/bin/minikube --- Disable swap for Kubernetes ---sudo swapoff -a --Ensure Docker service is runningsudo systemctl enable --now docker sudo systemctl status docker --no-pager || true --- cri-dockerd (use release tarball instead of git + go build) -----Download latest release tarball and install CLIcurl -LO https://github.com/Mirantis/cri-dockerd/releases/latest/download/cri-dockerd-linux-amd64.tar.gz sudo mkdir -p /usr/local/bin sudo tar -xvf cri-dockerd-linux-amd64.tar.gz sudo install -o root -g root -m 0755 cri-dockerd /usr/local/bin/cri-dockerd rm -f cri-dockerd-linux-amd64.tar.gz cri-dockerd--2. Install systemd units from project packagingcurl -LO https://raw.githubusercontent.com/Mirantis/cri-dockerd/master/packaging/systemd/cri-docker.service curl -LO https://raw.githubusercontent.com/Mirantis/cri-dockerd/master/packaging/systemd/cri-docker.socket sudo cp cri-docker.service cri-docker.socket /etc/systemd/system/ --Point unit file to /usr/local/bin/cri-dockerdsudo sed -i -e 's,/usr/bin/cri-dockerd,/usr/local/bin/cri-dockerd,' /etc/systemd/system/cri-docker.service --Reload and enable servicessudo systemctl daemon-reload sudo systemctl enable cri-docker.service sudo systemctl enable --now cri-docker.socket sudo systemctl status cri-docker.socket --no-pager || true --(Optional) kernel protection toggle (only if your environment requires it)sudo sysctl fs.protected_regular=0--- crictl (use official GitHub releases) -----Pin to the version you need. v1.25.0 matches your original lab; update if your Kubernetes minor differs.CRICTL_VERSION="v1.25.0" curl -LO https://github.com/kubernetes-sigs/cri-tools/releases/download/$%7BCRICTL_VERSION%7D/crictl-$%7BCRICTL_VERSION%7D-linux-amd64.tar.gz sudo tar zxvf crictl-${CRICTL_VERSION}-linux-amd64.tar.gz -C /usr/local/bin rm -f crictl-${CRICTL_VERSION}-linux-amd64.tar.gz --- Minikube start (driver none) hooking cri-dockerd ---Use the Docker runtime with cri-dockerd socket, matching your lab approachsudo -E minikube start --driver=none --container-runtime=docker --cri-socket=unix:///var/run/cri-dockerd.sock --- Calico CNI (Typha) for node Ready state ---kubectl apply -f https://docs.projectcalico.org/manifests/calico-typha.yaml echo "All done. Verify with: kubectl get nodes && kubectl get pods -A"3. note after the 1st replaced wget, the chmod 755 is done on a different file.
